CREATE TABLE MEMBER_INFO(
	MEMBER_IDX NUMBER PRIMARY KEY,
	MEMBER_NAME VARCHAR2(50) NOT NULL,
	MEMBER_NICKNAME VARCHAR2(50) NOT NULL UNIQUE,
	MEMBER_EMAIL VARCHAR2(100) NOT NULL UNIQUE,
	MEMBER_PWD VARCHAR2(100) NOT NULL
);

CREATE TABLE BOARD_INFO(
    BOARD_IDX NUMBER PRIMARY KEY,
    BOARD_TITLE VARCHAR2(100) NOT NULL, -- 제목
    BOARD_CONTENT VARCHAR2(1000) NOT NULL, -- 내용
    BOARD_REGDATE DATE DEFAULT SYSDATE NOT NULL, -- 작성일, DATE 타입으로 변경
    BOARD_DEPTH VARCHAR2(1000), -- 댓글
    BOARD_FILENAME VARCHAR2(300),
    MEMBER_IDX NUMBER,
    MEMBER_NICKNAME VARCHAR2(50),
    CONSTRAINT MEMBER_IDX_FK FOREIGN KEY (MEMBER_IDX) REFERENCES MEMBER_INFO(MEMBER_IDX),
    CONSTRAINT MEMBER_NICKNAME_FK FOREIGN KEY (MEMBER_NICKNAME) REFERENCES MEMBER_INFO(MEMBER_NICKNAME)
);
